53  template<class T> void addtoendof(vQueue &q) {
54 
55  //the bottle is stored as TAG (EVENTS) TAG (EVENTS)
56  for(int i = 0; i < Bottle::size(); i+=2) {
57 
58  //so for each TAG we create an event of that type
59  vEvent * e = emorph::createEvent(Bottle::get(i).asString());
60  if(!e) {
61  std::cerr << "Warning: could not get bottle type during vBottle::"
62  "get<>(). Check vBottle integrity." << std::endl;
63  continue;
64  }
65 
66  //and if e is of type T we can continue to get the events
67  if(!dynamic_cast<T*>(e)) continue;
68 
69  //we get the (EVENTS)
70  Bottle * b = Bottle::get(i+1).asList();
71  if(!b) {
72  std::cerr << "Warning: could not get event data as a list after "
73  "getting correct tag (e.g. AE) in vBottle::getAll(). "
74  "Check vBottle integrity" << std::endl;
75  delete(e);
76  break;
77  }
78 
79  //and decode each one also creating the memory with clone
80  int pos = 0;
81  while(pos < b->size()) {
82  if(e->decode(*b, pos)) {
83  q.push_back(e->clone());
84  }
85  }
86 
87  //finally we don't need the last event
88  delete(e);
89 
90  }
91 
92  }

172 class vBottleMimic : public yarp::os::Portable {

174 private:

182  const static unsigned int MINELSZ = sizeof(u_int32_t) * 2;
183 
184 public:
185 
186  vBottleMimic() {
187  header1.push_back(BOTTLE_TAG_LIST); //bottle code
188  header1.push_back(2); //elements in bottle "AE" then bottle data
189  header1.push_back(BOTTLE_TAG_STRING); //code for string
190  header1.push_back(2); // length of string
191  header2.push_back('A');
192  header2.push_back('E');
193  header3.push_back(BOTTLE_TAG_LIST|BOTTLE_TAG_INT); // bottle code + specialisation with ints
194  header3.push_back(0); // <- set the number of ints here (2 * #v's)
195  }
196 
197  void setdata(const char * datablock, unsigned int datalength) {
198  header3[1] = 2 * (datalength / MINELSZ); //forced to be x8
199  this->datablock = datablock;
200  this->datalength = MINELSZ * header3[1] / 2; //forced to be x8
201 
202  }
203 
204  virtual bool read(yarp::os::ConnectionReader& connection) {
205  return false;
206  }

208  virtual bool write(yarp::os::ConnectionWriter& connection) {
209 
210  connection.appendExternalBlock((const char *)header1.data(),
211  header1.size() * sizeof(u_int32_t));
212  connection.appendExternalBlock((const char *)header2.data(),
213  header2.size() * sizeof(char));
214  connection.appendExternalBlock((const char *)header3.data(),
215  header3.size() * sizeof(u_int32_t));
216  connection.appendExternalBlock(datablock, datalength);
